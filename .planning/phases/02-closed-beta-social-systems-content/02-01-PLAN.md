       ```luau
       local RateLimit = {}
       local lastCallAt: { [string]: { [number]: number } } = {}

       function RateLimit.check(player, remoteName, intervalSeconds)
           if type(player) ~= "table" and typeof(player) ~= "Instance" then return false end
           local userId = player.UserId
           if type(userId) ~= "number" then return false end
           if type(remoteName) ~= "string" or type(intervalSeconds) ~= "number" then return false end
           local now = os.clock()
           local bucket = lastCallAt[remoteName]
           if not bucket then bucket = {}; lastCallAt[remoteName] = bucket end
           local last = bucket[userId]
           if last and (now - last) < intervalSeconds then
               return false
           end
           bucket[userId] = now
           return true
       end

       function RateLimit._reset()
           lastCallAt = {}
       end

       return RateLimit
       ```
